1. Get the Dask Scheduler connection string by running this commands in the same shell:

  NOTE: It may take a few minutes for the LoadBalancer IP to be available, until that the commands below will not work.  
  You can watch the status by running 'kubectl get svc --namespace {{ .Release.Namespace }} -w {{ template "dask.scheduler-fullname" . }}'

  export DASK_SCHEDULER=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ template "dask.scheduler-fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$DASK_SCHEDULER:{{ .Values.scheduler.servicePort }}

2. Get the Dask Scheduler Web UI by running this commands in the same shell:

  NOTE: It may take a few minutes for the LoadBalancer IP to be available, until that the commands below will not work.  
  You can watch the status by running 'kubectl get svc --namespace {{ .Release.Namespace }} -w {{ template "dask.scheduler-fullname" . }}'

  export DASK_SCHEDULER_UI_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ template "dask.scheduler-fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$DASK_SCHEDULER_UI_IP:{{ .Values.webUI.servicePort }}
